package com.company;

import java.util.Scanner;

public class Main {


    public static void main(String[] args) {

        System.out.println("Bienvenido a Tacos Mike");
        int select = 0;
        int[] tTacos = new int[4];
        int[] tBebidas = new int[4];
        int totalTacos = 0;
        int totalBebidas = 0;
        int totalCuenta = 0;


        while (select == 1 || select == 2 || select == 0 || select == 3) {
            switch (select) {
                case 0:
                    select = menu();
                    break;
                case 1:
                    tTacos = tacos();
                    select = 0;
                    break;
                case 2:
                    tBebidas = bebidas();
                    select = 0;
                    break;
                case 3:
                    totalTacos = (tTacos[0] * 10) + (tTacos[1] * 8) + (tTacos[2] * 7) + (tTacos[3] * 9);
                    totalBebidas = (tBebidas[0] * 11) + (tBebidas[1] * 12) + (tBebidas[2] * 13) + (tBebidas[3] * 13);
                    totalCuenta = totalBebidas + totalTacos;
                    System.out.println("El total de tacos es = " + totalTacos);
                    System.out.println("El total de bebidas es = " + totalBebidas);
                    System.out.println("El total de tu cuenta es = " + totalCuenta);
                    select = 4;
                    break;


            }


        }


    }

    public static int menu() {
        System.out.println("Menu" + "\n" + "Para tacos presiona 1" + "\n" + "Para bebidas presiona 2" + "\n" + "Para el total de tu orden presiona 3" + "\n" + "Para cancelar tu orden presiona 4");
        Scanner input = new Scanner(System.in);
        int opcion = input.nextInt();

        return opcion;
    }

    public static int[] tacos() {
        int topcion = 0;
        int tcantidad = 0;
        int tAsada = 0;
        int tPastor = 0;
        int tChorizo = 0;
        int tTripa = 0;
        boolean flag = true;
        int[] ar = new int[4];

        ar[0] = tAsada;
        ar[1] = tPastor;
        ar[2] = tChorizo;
        ar[3] = tTripa;

        System.out.println("Menu de tacos" + "\n" + "Para Asada presiona 1" + "\n" + "Para Pastor presiona 2" + "\n" + "Para Chorizo presiona 3" + "\n" + "Para Tripa presiona 4" + "\n" + "Terminar orden presiona 5");
        while (flag != false) {
            System.out.println("¿Cual tipo de taco quieres?");
            Scanner input = new Scanner(System.in);
            topcion = input.nextInt();

            if (topcion == 1) {
                System.out.println("¿Cuantos quieres de Asada?");
                tcantidad = input.nextInt();
                ar[0] = ar[0] + tcantidad;
            }
            if (topcion == 2) {
                System.out.println("¿Cuantos quieres de Pastor?");
                tcantidad = input.nextInt();
                ar[1] = ar[1] + tcantidad;
            }
            if (topcion == 3) {
                System.out.println("¿Cuantos quieres de Chorizo?");
                tcantidad = input.nextInt();
                ar[2] = ar[2] + tcantidad;
            }
            if (topcion == 4) {
                System.out.println("¿Cuantos quieres de Tripa?");
                tcantidad = input.nextInt();
                ar[3] = ar[3] + tcantidad;
            }
            if (topcion == 5) {
                System.out.println("Asada:" + ar[0]);
                System.out.println("Pastor:" + ar[1]);
                System.out.println("Chorizo:" + ar[2]);
                System.out.println("Tripa:" + ar[3]);
                flag = false;
            }

        }

        return ar;
    }

    public static int[] bebidas() {
        int bopcion = 0;
        int bcantidad = 0;
        int bCoca = 0;
        int bSquirt = 0;
        int bJamaica = 0;
        int bHorchata = 0;
        int[] ar = new int[4];
        ar[0] = bCoca;
        ar[1] = bSquirt;
        ar[2] = bJamaica;
        ar[3] = bHorchata;

        System.out.println("Menu de Bebidas" + "\n" + "Para Coca presiona 1" + "\n" + "Para Squirt presiona 2" + "\n" + "Para Jamaica presiona 3" + "\n" + "Para Horchata presiona 4" + "\n" + "Terminar orden presiona 5");
        while (bopcion != 5) {
            System.out.println("¿Cual tipo de Bebida quieres?");
            Scanner input = new Scanner(System.in);
            bopcion = input.nextInt();

            switch (bopcion) {
                case 1:
                    System.out.println("¿Cuantas Cocas quieres?");
                    bcantidad = input.nextInt();
                    ar[0] = ar[0] + bcantidad;
                    break;

                case 2:
                    System.out.println("¿Cuantos Squirt quieres?");
                    bcantidad = input.nextInt();
                    ar[1] = ar[1] + bcantidad;
                    break;

                case 3:
                    System.out.println("¿Cuantas  Jamaicas quieres?");
                    bcantidad = input.nextInt();
                    ar[2] = ar[2] + bcantidad;
                    break;

                case 4:
                    System.out.println("¿Cuantas Horchatas quieres?");
                    bcantidad = input.nextInt();
                    ar[3] = ar[3] + bcantidad;
                    break;

                case 5:


                    System.out.println("Cocas:" + ar[0]);
                    System.out.println("Squirts:" + ar[1]);
                    System.out.println("Jamaicas:" + ar[2]);
                    System.out.println("Horchatas:" + ar[3]);
                    break;


            }
        }


        return ar;
    }

}
